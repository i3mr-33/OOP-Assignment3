269template <typename T>
270class GameManager {
271 Board<T>* boardPtr;
272 Player<T>* players[2];
273 UI<T>* ui;
274
275public:
279 GameManager(Board<T>* b, Player<T>* p[2], UI<T>* u)
280 : boardPtr(b), ui(u) {
281 players[0] = p[0];
282 players[1] = p[1];
283 players[0]->set_board_ptr(b);
284 players[1]->set_board_ptr(b);
285 }
286
290 void run() {
291 ui->display_board_matrix(boardPtr->get_board_matrix());
292 Player<T>* currentPlayer = players[0];
293
294 while (true) {
295 for (int i : {0, 1}) {
296 currentPlayer = players[i];
297 Move<T>* move = ui->get_move(currentPlayer);
298
299 while (!boardPtr->update_board(move))
300 move = ui->get_move(currentPlayer);
301
302 ui->display_board_matrix(boardPtr->get_board_matrix());
303
304 if (boardPtr->is_win(currentPlayer)) {
305 ui->display_message(currentPlayer->get_name() + " wins!");
306 return;
307 }
308 if (boardPtr->is_lose(currentPlayer)) {
309 ui->display_message(players[1 - i]->get_name() + " wins!");
310 return;
311 }
312 if (boardPtr->is_draw(currentPlayer)) {
313 ui->display_message("Draw!");
314 return;
315 }
316 }
317 }
318 }
319};
